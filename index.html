	document.body.innerHTML = timestamp;
	rAF(step);

transitionend event

Fired when a CSS transition has completed. In the case where a transition is removed before completion, such as if the transition-property is removed, then the event will not fire.


var element = document.querySelector('.transition');

function showMessage(e) {
	console.log(e);
}

element.addEventListener("transitionend", showMessage);
